# FormByLjx-表单

编写人：刘崛雄

创建时间：2022.04.19

最近更新：2022.04.19

带收缩功能的页面提示。

何时使用

当某个页面需要展示功能操作或业务相关信息时。

<template>
  <div style="witdth=100%">
    <!-- 布局选项 -->
    <a-row style="margin-top:20px">
      <a-radio-group :value="layout" @change="radioChange">
        <a-radio-button value="horizontal">水平布局</a-radio-button>
        <a-radio-button value="vertical">垂直布局</a-radio-button>
      </a-radio-group>
    </a-row>
    <!-- 操作按钮 -->
    <a-row style="margin-top:20px">
      <a-space>
        <a-button type="danger" @click="lock">锁定</a-button>
        <a-button type="primary" @click="setEditable">编辑</a-button>
        <a-button type="primary" @click="setValue">设置值</a-button>
        <a-button type="primary" @click="reset">重置</a-button>
        <a-button type="primary" @click="clear">清空</a-button>
        <a-button type="primary" @click="onSubmit">提交</a-button>
      </a-space>
    </a-row>

    <!-- 表单 -->
    <div style="width:1000px;margin-top:20px">
      <FormByLjx
        ref="testForm"
        style=" background-color: #fafafa;"
        :layout="layout"
        :items="getItems"
        :editable="false"
        @change="onChange"
      >
        <!-- 插槽--评分 -->
        <template slot="rate">
          <a-rate v-model="rateValue" allow-half />
        </template>
      </FormByLjx>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      layout: 'horizontal',
      flag: true,
      rateValue: 1
    }
  },
  computed: {
    getLayout() {
      return this.layout
    },
    getItems() {
      return [// 表单-配置项，放入computed属性后，可根据配置项相关值变化，动态渲染form表单变化，此处通过flag控制【备注】的可编辑状态
        { span: 6, type: 'text', labelText: '姓名', fieldName: 'name', required: true, initialValue: '无名氏' },
        { span: 6, type: 'select', labelText: '性别', fieldName: 'sex', required: true, initialValue: 2,
          options: [{ key: 0, label: '男' }, { key: 1, label: '女' }, { key: 2, label: '未知' }], showSearch: true
        },
        { span: 6, type: 'number', labelText: '年龄', fieldName: 'age', required: true, initialValue: 0, min: 3, max: 80 },
        { span: 12, type: 'select', labelText: '爱好', fieldName: 'hobby', required: true, initialValue: { value: 3, label: '跳舞' }, mode: 'multiple', showSearch: true, labelInValue: true,
          options: [{ key: 0, label: '游戏' }, { key: 1, label: '睡觉' }, { key: 2, label: '唱歌' }, { key: 3, label: '跳舞' }, { key: 4, label: '美食' }],
          labelCol: { span: 4 }, wrapperCol: { span: 20 }
        },
        { span: 6, type: 'checkboxGroup', labelText: '常吃水果', fieldName: 'fruits', required: true, options: [{ label: '苹果', value: 'apple' }, { label: '梨', value: 'pear' }] },
        { span: 6, type: 'radioGroup', labelText: '单选', fieldName: 'radioGroup', required: true, options: [{ label: '买', value: 'buy' }, { label: '卖', value: 'sale' }] },
        { span: 6, type: 'switch', labelText: '开关', fieldName: 'switch' },
        { span: 6, type: 'select', labelText: '体型', fieldName: 'bodyType', required: true, initialValue: { key: 3, label: '瘦' }, labelInValue: true,
          options: [{ key: 0, label: '高' }, { key: 1, label: '矮' }, { key: 2, label: '胖' }, { key: 3, label: '瘦' }]
        },
        { span: 6, type: 'date', labelText: '出生日期', fieldName: 'birthday', required: true, initialValue: '2022-04-05' },
        { span: 6, type: 'time', labelText: '出生时间', fieldName: 'time', required: true, initialValue: '8:00' },

        { span: 12, type: 'dateRange', labelText: '起止日期', fieldName: 'timeRange', required: true,
          initialValue: ['2022-04-06', '2022-04-14'], labelCol: { span: 4 }, wrapperCol: { span: 20 }
        },
        { span: 12, type: 'other', labelText: '评分', fieldName: 'rate' },
        { span: 24, type: 'text', textRows: 6, labelText: '备注', fieldName: 'beizhu', initialValue: 'sssssss', disabled: this.flag,
          labelCol: { span: 4 }, wrapperCol: { span: 20 }
        }
      ]
    }
  },
  methods: {
    radioChange(e) {
      this.layout = e.target.value
      console.log('radioChange--e', e)
    },

    setEditable() { // 设置可编辑
      this.$refs.testForm.formEditable = true
    },
    lock() { // 锁定
      this.$refs.testForm.formEditable = false
    },
    setValue() { // 表单项设置值
      this.$refs.testForm.formObject.setFieldsValue({ 'name': 'xxx' })
      this.$refs.testForm.formObject.setFieldsValue({ 'sex': 0 })
      this.$refs.testForm.formObject.setFieldsValue({ 'age': 27 })
      this.$refs.testForm.formObject.setFieldsValue({ 'hobby': [{ key: 0, label: '游戏' }, { key: 1, label: '睡觉' }, { key: 2, label: '唱歌' }] })
      this.$refs.testForm.formObject.setFieldsValue({ 'fruits': ['pear', 'apple'] })
      this.$refs.testForm.formObject.setFieldsValue({ 'switch': true })
      this.$refs.testForm.formObject.setFieldsValue({ 'radioGroup': 'buy' })
      this.$refs.testForm.formObject.setFieldsValue({ 'bodyType': { key: 2, label: '胖' }})
      this.$refs.testForm.formObject.setFieldsValue({ 'birthday': '1995-1-21' })
      this.$refs.testForm.formObject.setFieldsValue({ 'time': '18:00' })
      this.$refs.testForm.formObject.setFieldsValue({ 'timeRange': ['1995-01-21', '2082-04-14'] })
      this.$refs.testForm.formObject.setFieldsValue({ 'beizhu': '测试' })

      this.rateValue = 4
    },
    clear() { // 表单清空
      this.$refs.testForm.formObject.setFieldsValue({ 'name': undefined })
      this.$refs.testForm.formObject.setFieldsValue({ 'age': undefined })
      this.$refs.testForm.formObject.setFieldsValue({ 'sex': undefined })
      this.$refs.testForm.formObject.setFieldsValue({ 'hobby': undefined })
      this.$refs.testForm.formObject.setFieldsValue({ 'fruits': [] })
      this.$refs.testForm.formObject.setFieldsValue({ 'switch': false })
      this.$refs.testForm.formObject.setFieldsValue({ 'radioGroup': undefined })

      this.$refs.testForm.formObject.setFieldsValue({ 'bodyType': undefined })
      this.$refs.testForm.formObject.setFieldsValue({ 'birthday': undefined })
      this.$refs.testForm.formObject.setFieldsValue({ 'time': undefined })
      this.$refs.testForm.formObject.setFieldsValue({ 'timeRange': undefined })
      this.$refs.testForm.formObject.setFieldsValue({ 'beizhu': undefined })
      this.rateValue = 0
    },
    onSubmit() { // 提交
      this.$refs.testForm.formObject.validateFields((errors, values) => {
        if (!errors) { // 校验成功
          console.log('校验成功--values', values)
        } else {
          console.log('校验失败--errors', errors)
        }
      })
    },
    reset() { // 重置表格
      this.$refs.testForm.formObject.resetFields()
      this.rateValue = 1
    },
    onChange(filedName, value, allValues) { // 监听表单变化
      console.log('onChange---filedName', filedName)
      console.log('onChange---value', value)
      console.log('onChange---allValues', allValues)
      if (filedName === 'switch') {
        if (value) {
          this.flag = false
        } else {
          this.flag = true
        }
      }
    }
  }
}
</script>
